          using Experience.Teams.TeamMember;

          using McMasterUniversity.Education.Engineering;

          /// <summary>
          /// Quick Resume implementation
          /// </summary>
          public class Resume
            : IResume
          {
            public CustomSoftware Hire(Client you)
            {
              IRequirement[] customRequirements = you.GetRequirements();

              var software = SoftwareDevelopmentExperience
                .Analyze(customRequirements)
                .ProvideFeedback(you)
                .Build();

              Operations.Setup(software);
              QualityAssurance.Test(software);
              UserAcceptanceTesting.Test(software);

              return software;
            }
          }
